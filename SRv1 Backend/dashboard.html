const express = require("express"); const app = express(); // Serve static files from the public directory app.use(express.static("public")); // Handle POST request to create a user account app.post("/api/user", (req, res) => { const { username, password } = req.body; // Save the username and password to a database or file // Send a response indicating success or failure res.status(200).json({ message: "User account created successfully" }); }); // Handle GET request to fetch user data app.get("/api/user", (req, res) => { const { username, password } = req.query; // Look up the user data in a database or file // Send the user data in the response res.status(200).json({ username, password }); }); // Start the server const port = 3000; app.listen(port, () => { console.log(`Server listening on port ${port}`); });
